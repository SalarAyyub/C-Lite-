#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, max, c=0;
    cin>>n;

    int a[n];

    for(int i=0; i<n; i++)
    {
        cin>>a[i];
    }

    max = *max_element(a, a+n);

    for(int i=0; i<n; i++)
    {
        if(a[i] == max)
        {
            c++;
        }
    }

    cout<<c;

    return 0;
}
