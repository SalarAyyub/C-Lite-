#include <bits/stdc++.h>

using namespace std;

int main()
{
    long n, min, min2;
    cin>>n;

    long a[n];

    if(n <= 2)
    {
        cout<<"Insufficient elements.";
    }
    else
    {
        for(int i=0; i<n; i++)
        {
            cin>>a[i];
        }

        min = min2 = INT_MAX;

        for(int i=0; i<n; i++)
        {
            if(min > a[i])
            {
                min2 = min;
                min = a[i];
            }
            else if(a[i] < min2 && a[i] != min)
            {
                min2 = a[i];
            }
        }

        if(min2 == INT_MAX)
        {
            cout<<"No second smallest element in the array.";
        }
        else
        {
            cout<<min<<" "<<min2;
        }

    }

    return 0;
}
