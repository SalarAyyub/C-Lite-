#include <bits/stdc++.h>

using namespace std;

int theThaw(int a[], int n)
{
    int est=0, count=0;

    for(int i=0; i<n; i++)
    {
        if(a[i] > 0)
        {
            count++;
            est = max(est,count);
        }
        else
        {
            count=0;
        }
    }

    return est;
}

int main()
{
    int n, c=0, thaw;
    cin>>n;

    int a[n];

    for(int i=0; i<n; i++)
    {
        cin>>a[i];
    }

    thaw = theThaw(a, n);

    cout<<thaw;


    return 0;
}
