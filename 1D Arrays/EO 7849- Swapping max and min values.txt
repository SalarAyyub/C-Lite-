#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin>>n;

    int a[n];

    for(int i=0; i<n; i++)
    {
        cin>>a[i];
    }

    int largest = *max_element(a, a+n);
    int smallest = *min_element(a,a+n);


    for(int i=0; i<n; i++)
    {
        if(a[i] == largest)
        {
            cout<<smallest<<" ";
        }
        else if(a[i] == smallest)
        {
            cout<<largest<<" ";
        }
        else
        {
            cout<<a[i]<<" ";
        }
    }

    return 0;
}
