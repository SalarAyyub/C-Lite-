#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t;

    while(t--)
    {
        int total = 0, majority = -1, id = 0, num = 0, n;
        cin >> n;

        for(int i = 1; i <= n; ++i)
        {
            int x;
            cin >> x;

            total += x;

            if(x > majority)
            {
                majority = x;
                id = i;
                num = 1;
            }
            else if(x == majority)
            {
                ++num;
            }
        }

        if(num > 1)
        {
            cout<<"no winner"<<endl;
        }
        else if(2*majority <= total)
        {
            cout <<"minority winner "<<id<<endl;
        }
        else
        {
            cout <<"majority winner "<<id<<endl;
        }

    }

    return 0;
}
