#include <bits/stdc++.h>

using namespace std;

int main() {
    
    int n, r;
    
    while(cin >> n >> r && n && r) {
        int card[n];
        for (int i = 0; i < n; ++i) {
            card[i] = n - i;
        }
        for (int i = 0; i < r; ++i) {
            int p, c;
            cin >> p >> c;
            rotate(card, card + p - 1, card + p + c - 1);
        }
        cout << card[0] << endl;
    }
    return 0;
}
